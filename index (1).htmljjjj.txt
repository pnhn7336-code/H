// ================= WALLET BALANCE =================
async function fetchWalletBalance() {
  try {
    const apiKey = settings.apiKey, secretKey = settings.secretKey;
    if (!apiKey || !secretKey) {
      document.getElementById('walletStatus').textContent = 'No API';
      document.getElementById('walletStatus').className = 'card-badge badge-red';
      return;
    }
    const timestamp = Date.now();
    const queryString = `timestamp=${timestamp}`;
    const signature = generateSignature(queryString, secretKey);
    const response = await fetch(`${getBaseUrl()}/fapi/v2/account?${queryString}&signature=${signature}`, {
      headers: { 'X-MBX-APIKEY': apiKey }
    });
    if (!response.ok) {
      document.getElementById('walletStatus').textContent = 'Error';
      document.getElementById('walletStatus').className = 'card-badge badge-red';
      return;
    }
    const data = await response.json();
    walletData = data;
    const usdtAsset = data.assets.find(a => a.asset === 'USDT');
    if (usdtAsset) {
      const totalBalance = parseFloat(usdtAsset.walletBalance);
      const available = parseFloat(usdtAsset.availableBalance);
      const marginUsed = parseFloat(usdtAsset.initialMargin) + parseFloat(usdtAsset.maintMargin);
      const unrealizedPnL = parseFloat(usdtAsset.unrealizedProfit);
      const maintMargin = parseFloat(usdtAsset.maintMargin);
      const riskRatio = totalBalance > 0 ? (marginUsed / totalBalance) * 100 : 0;
      document.getElementById('walletBalance').textContent = '$' + totalBalance.toFixed(2);
      document.getElementById('walletAvailable').textContent = '$' + available.toFixed(2);
      document.getElementById('walletMargin').textContent = '$' + marginUsed.toFixed(2);
      document.getElementById('walletUnrealized').textContent = (unrealizedPnL >= 0 ? '+' : '') + '$' + unrealizedPnL.toFixed(2);
      document.getElementById('walletUnrealized').style.color = unrealizedPnL >= 0 ? 'var(--accent)' : 'var(--accent-red)';
      document.getElementById('walletMaintMargin').textContent = '$' + maintMargin.toFixed(2);
      document.getElementById('walletRiskRatio').textContent = riskRatio.toFixed(1) + '%';
      document.getElementById('walletRiskRatio').style.color = riskRatio > 80 ? 'var(--accent-red)' : riskRatio > 50 ? 'var(--accent-gold)' : 'var(--accent)';

function updateMoneyManagement(balance) {
  const actualBalance = balance || (walletData ? parseFloat(walletData.assets.find(a => a.asset === 'USDT')?.walletBalance || 0) : settings.baseCapital);
  const maxRisk = actualBalance * (settings.riskPerTrade / 100);
  const maxDailyLoss = actualBalance * (settings.maxDailyLoss / 100);
  const currentLoss = dailyLoss * (actualBalance / 100);
  const remainingBudget = maxDailyLoss - currentLoss;
  const winRate = stats.winRate / 100 || 0.5;
  const avgWin = trades.filter(t => t.roi > 0).reduce((sum, t) => sum + t.roi, 0) / (stats.wins || 1) || 5;
  const avgLoss = trades.filter(t => t.roi <= 0).reduce((sum, t) => sum + Math.abs(t.roi), 0) / (stats.losses || 1) || 2;
  const kelly = winRate > 0 && avgWin > 0 ? ((winRate * avgWin) - ((1 - winRate) * avgLoss)) / avgWin : 0.02;
  let kellyPosition = actualBalance * Math.max(0, kelly) * settings.kellyFraction;
  if (settings.antiMartingale && consecutiveLosses > 0) {
    kellyPosition = kellyPosition * Math.pow(0.8, consecutiveLosses);
  }
  const recLeverage = settings.autoLeverage ? Math.min(settings.maxLeverage, Math.max(5, Math.floor(20 / (avgLoss || 2)))) : settings.leverage;
  const safetyBuffer = actualBalance > 0 ? ((actualBalance - currentLoss) / actualBalance * 100).toFixed(1) + '%' : 'N/A';
  document.getElementById('mmBalance').textContent = '$' + actualBalance.toFixed(2);
  document.getElementById('mmMaxRisk').textContent = '$' + maxRisk.toFixed(2);
  document.getElementById('mmMaxDailyLoss').textContent = '$' + maxDailyLoss.toFixed(2);
  document.getElementById('mmCurrentLoss').textContent = '$' + currentLoss.toFixed(2);
  document.getElementById('mmRemainingBudget').textContent = '$' + remainingBudget.toFixed(2);
  document.getElementById('mmPositionSize').textContent = '$' + kellyPosition.toFixed(2);
  document.getElementById('mmRecLeverage').textContent = recLeverage + 'x';
  document.getElementById('mmSafetyBuffer').textContent = safetyBuffer;
  document.getElementById('mmSafetyBuffer').style.color = remainingBudget > maxDailyLoss * 0.5 ? 'var(--accent)' : 'var(--accent-red)';
}

async function analyzeMarketTrend() {

      const ema50 = calculateEMA(closes, 50);
      const currentPrice = closes[closes.length - 1];
      const prevPrice = closes[closes.length - 2];
      const change24h = ((currentPrice - closes[0]) / closes[0]) * 100;

// ================= TECHNICAL ANALYSIS =================
function calculateRSI(closes, period = 14) {
  if (closes.length < period + 1) return 50;
  let gains = 0, losses = 0;
  for (let i = 1; i <= period; i++) {
    const change = closes[closes.length - i] - closes[closes.length - i - 1];
    if (change >= 0) gains += change; else losses -= change;
  }
  const avgGain = gains / period, avgLoss = losses / period;
  if (avgLoss === 0) return 100;
  const rs = avgGain / avgLoss;
  return 100 - (100 / (1 + rs));
}

function calculateEMA(closes, period) {
  const k = 2 / (period + 1);
  let ema = closes.slice(0, period).reduce((a, b) => a + b, 0) / period;
  for (let i = period; i < closes.length; i++) { ema = closes[i] * k + ema * (1 - k); }
  return ema;
}

function calculateBB(closes, period = 20, stdDev = 2) {
  const sma = closes.slice(-period).reduce((a, b) => a + b, 0) / period;
  const variance = closes.slice(-period).reduce((sum, price) => sum + Math.pow(price - sma, 2), 0) / period;
  const std = Math.sqrt(variance);
  return { upper: sma + stdDev * std, middle: sma, lower: sma - stdDev * std };
}

function calculateATR(klines, period = 14) {
  if (klines.length < period + 1) return 0;
  let trSum = 0;
  for (let i = klines.length - period; i < klines.length; i++) {
    const high = parseFloat(klines[i][2]), low = parseFloat(klines[i][3]), prevClose = parseFloat(klines[i-1][4]);
    trSum += Math.max(high - low, Math.abs(high - prevClose), Math.abs(low - prevClose));
  }
  return trSum / period;
}

function calculateVolumeRatio(klines) {
  if (klines.length < 20) return 1;
  const currentVol = parseFloat(klines[klines.length - 1][5]);
  const avgVol = klines.slice(-20).reduce((sum, k) => sum + parseFloat(k[5]), 0) / 20;
  return currentVol / avgVol;
}

function detectKeyLevels(klines) {
  const highs = klines.map(k => parseFloat(k[2])), lows = klines.map(k => parseFloat(k[3]));
  return { resistance: Math.max(...highs.slice(-20)), support: Math.min(...lows.slice(-20)) };
}

function calculateLiquidationPrice(entry, direction, leverage) {
  const liqDistance = (1 / leverage) * 0.9;
  return direction === 'LONG' ? entry * (1 - liqDistance) : entry * (1 + liqDistance);
}

function calculatePositionSize(balance, riskPercent, entry, stopLoss, leverage) {
  const riskAmount = balance * (riskPercent / 100);
  const priceDiff = Math.abs(entry - stopLoss);
  const stopLossPercent = priceDiff / entry;
  if (stopLossPercent === 0) return 0;
  const positionSize = riskAmount / stopLossPercent;
  const maxPosition = balance * leverage * 0.90;
  return Math.min(positionSize, maxPosition);
}

// ================= SNIPER SIGNAL ENGINE =================
async function analyzeSymbol(symbol) {
  try {
    const klines5m = await getKlines(symbol, '5m', 100);
    const klines1h = await getKlines(symbol, '1h', 50);
    if (!klines5m || !klines1h) return null;
    const closes5m = klines5m.map(k => parseFloat(k[4]));
    const currentPrice = closes5m[closes5m.length - 1];
    const rsi = calculateRSI(closes5m, 14);
    const ema20 = calculateEMA(closes5m, 20);
    const ema50 = calculateEMA(closes5m, 50);
    const bb = calculateBB(closes5m, 20, 2);
    const atr = calculateATR(klines5m, 14);
    const volRatio = calculateVolumeRatio(klines5m);
    const levels = detectKeyLevels(klines5m);
    const closes1h = klines1h.map(k => parseFloat(k[4]));
    const ema20_1h = calculateEMA(closes1h, 20);

    let confluence = 0, direction = null, reasons = [];
    if (settings.useRSIFilter) {
      if (rsi < settings.minRSI) { direction = 'LONG'; confluence += 20; reasons.push('RSI Oversold'); }
      else if (rsi > settings.maxRSI) { direction = 'SHORT'; confluence += 20; reasons.push('RSI Overbought'); }
    }
    if (currentPrice > ema20 && ema20 > ema50) {
      if (!direction) direction = 'LONG';
      if (direction === 'LONG') { confluence += 15; reasons.push('EMA Bullish'); }
    } else if (currentPrice < ema20 && ema20 < ema50) {
      if (!direction) direction = 'SHORT';
      if (direction === 'SHORT') { confluence += 15; reasons.push('EMA Bearish'); }
    }
    if (currentPrice <= bb.lower && direction === 'LONG') { confluence += 15; reasons.push('BB Lower Touch'); }
    else if (currentPrice >= bb.upper && direction === 'SHORT') { confluence += 15; reasons.push('BB Upper Touch'); }
    if (volRatio >= settings.minVolumeRatio) { confluence += 10; reasons.push('High Volume'); }
    if (direction === 'LONG' && Math.abs(currentPrice - levels.support) / currentPrice < 0.005) { confluence += 15; reasons.push('Support Bounce'); }
    else if (direction === 'SHORT' && Math.abs(currentPrice - levels.resistance) / currentPrice < 0.005) { confluence += 15; reasons.push('Resistance Reject'); }
    if (settings.useMultiTF) {
      if (direction === 'LONG' && currentPrice > ema20_1h) { confluence += 10; reasons.push('HTF Bullish'); }
      else if (direction === 'SHORT' && currentPrice < ema20_1h) { confluence += 10; reasons.push('HTF Bearish'); }
    }
    const recentCloses = closes5m.slice(-10);
    let hh = 0, ll = 0;
    for (let i = 2; i < recentCloses.length; i++) {
      if (recentCloses[i] > recentCloses[i-1] && recentCloses[i-1] > recentCloses[i-2]) hh++;
      if (recentCloses[i] < recentCloses[i-1] && recentCloses[i-1] < recentCloses[i-2]) ll++;
    }
    if (direction === 'LONG' && hh > ll) { confluence += 10; reasons.push('Bullish Structure'); }
    else if (direction === 'SHORT' && ll > hh) { confluence += 10; reasons.push('Bearish Structure'); }

    if (confluence >= settings.minConfluence && direction) {
      const slDistance = atr * 1.5;
      const sl = direction === 'LONG' ? currentPrice - slDistance : currentPrice + slDistance;
      const rr = settings.rrTarget;
      const tpDistance = slDistance * rr;
      const tp1 = direction === 'LONG' ? currentPrice + (tpDistance * 0.2) : currentPrice - (tpDistance * 0.2);
      const tp2 = direction === 'LONG' ? currentPrice + (tpDistance * 0.5) : currentPrice - (tpDistance * 0.5);
      const tp3 = direction === 'LONG' ? currentPrice + tpDistance : currentPrice - tpDistance;
      const balance = await getBinanceBalance() || settings.baseCapital;
      const positionSize = calculatePositionSize(balance, settings.riskPerTrade, currentPrice, sl, settings.leverage);
      const margin = positionSize / settings.leverage;
      const liqPrice = calculateLiquidationPrice(currentPrice, direction, settings.leverage);
      const slToLiq = Math.abs(sl - liqPrice) / currentPrice;
      const minSafeDistance = (1 / settings.leverage) * (settings.liqSafetyMargin / 100);
      if (slToLiq < minSafeDistance) return null;
      return {
        symbol, direction, entry: currentPrice, sl, tp1, tp2, tp3,
        confidence: confluence >= 80 ? 'SNIPER' : confluence >= 70 ? 'HIGH' : 'MEDIUM',
        reason: reasons.join(' + '), rsi: rsi.toFixed(1),
        momentum: ((closes5m[closes5m.length - 1] - closes5m[closes5m.length - 5]) / closes5m[closes5m.length - 5] * 100).toFixed(2),
        timestamp: new Date().toISOString(), confluence, positionSize, margin, liqPrice, atr: atr.toFixed(4)
      };
    }
    return null;
  } catch (e) { console.error('Analysis error:', e); return null; }
}

// ================= ORDER PLACEMENT (REAL TRADING) =================
async function placeBinanceOrder(signal, balance) {
  try {
    const apiKey = settings.apiKey, secretKey = settings.secretKey;
    if (!apiKey || !secretKey) {
      showToast('❌ Binance API keys not configured!', 'error');
      showModal('settingsModal'); switchTab('tab-api');
      return null;
    }
    if (settings.tradingMode === 'paper') {
      return { orderId: 'PAPER_' + Date.now(), status: 'FILLED', avgPrice: signal.entry, executedQty: (signal.positionSize / signal.entry).toFixed(6) };
    }
    const side = signal.direction === 'LONG' ? 'BUY' : 'SELL';
    const quantity = (signal.positionSize / signal.entry).toFixed(getQuantityPrecision(signal.symbol));
    const timestamp = Date.now();
    const leverageQuery = `symbol=${signal.symbol}&leverage=${settings.leverage}&timestamp=${timestamp}`;
    const leverageSig = generateSignature(leverageQuery, secretKey);
    await fetch(`${getBaseUrl()}/fapi/v1/leverage?${leverageQuery}&signature=${leverageSig}`, {
      method: 'POST', headers: { 'X-MBX-APIKEY': apiKey }
    });
    const orderQuery = `symbol=${signal.symbol}&side=${side}&type=MARKET&quantity=${quantity}&timestamp=${Date.now()}`;
    const orderSig = generateSignature(orderQuery, secretKey);
    const response = await fetch(`${getBaseUrl()}/fapi/v1/order?${orderQuery}&signature=${orderSig}`, {
      method: 'POST', headers: { 'X-MBX-APIKEY': apiKey }
    });
    const data = await response.json();
    if (data.code) {
      showToast(`❌ Order failed: ${data.msg}`, 'error');
      return null;
    }
    await setStopLossTakeProfit(signal, data.orderId, quantity);
    return data;
  } catch (e) { console.error('Order error:', e); showToast('❌ Order placement failed!', 'error'); return null; }
}

async function setStopLossTakeProfit(signal, parentOrderId, quantity) {
  try {
    const apiKey = settings.apiKey, secretKey = settings.secretKey;
    if (!apiKey || !secretKey || settings.tradingMode === 'paper') return;
    const closeSide = signal.direction === 'LONG' ? 'SELL' : 'BUY';
    const pricePrec = getPricePrecision(signal.symbol);
    const slQuery = `symbol=${signal.symbol}&side=${closeSide}&type=STOP_MARKET&stopPrice=${signal.sl.toFixed(pricePrec)}&quantity=${quantity}&timestamp=${Date.now()}`;
    const slSig = generateSignature(slQuery, secretKey);
    await fetch(`${getBaseUrl()}/fapi/v1/order?${slQuery}&signature=${slSig}`, {
      method: 'POST', headers: { 'X-MBX-APIKEY': apiKey }
    });
    const tpQuery = `symbol=${signal.symbol}&side=${closeSide}&type=TAKE_PROFIT_MARKET&stopPrice=${signal.tp3.toFixed(pricePrec)}&quantity=${quantity}&timestamp=${Date.now()}`;
    const tpSig = generateSignature(tpQuery, secretKey);
    await fetch(`${getBaseUrl()}/fapi/v1/order?${tpQuery}&signature=${tpSig}`, {
      method: 'POST', headers: { 'X-MBX-APIKEY': apiKey }
    });
  } catch (e) { console.error('SL/TP error:', e); }
}

async function scanAndTrade() {
  if (!botRunning) return;
  const today = new Date().toISOString().split('T')[0];
  if (dailyCount.date !== today) { dailyCount = { count: 0, date: today }; dailyLoss = 0; localStorage.setItem('sniper_daily', JSON.stringify(dailyCount)); }
  if (dailyCount.count >= settings.maxDailySignals) { showToast('📅 Daily trade limit reached (' + settings.maxDailySignals + ')'); stopTradingBot(); return; }
  const now = Date.now();
  const cooldownMs = settings.cooldown * 60 * 1000;
  if (now - lastTradeTime < cooldownMs) return;
  const activePos = positions.filter(p => p.status === 'active' || p.status === 'tp1_hit' || p.status === 'tp2_hit');
  if (activePos.length >= settings.maxPositions) return;
  const balance = await getBinanceBalance() || settings.baseCapital;
  if (balance < settings.minBalance) { showToast('❌ Insufficient balance!', 'error'); stopTradingBot(); return; }
  const coins = settings.selectedCoins || ['BTCUSDT', 'ETHUSDT'];
  let bestSignal = null, bestConfluence = 0;
  for (const symbol of coins) {
    try {
      const signal = await analyzeSymbol(symbol);
      if (signal && signal.confluence > bestConfluence) { bestSignal = signal; bestConfluence = signal.confluence; }
    } catch (e) { console.error('Scan error for', symbol, e); }
  }
  if (bestSignal) {
    updateLiquidationDisplay(bestSignal);
        // Demo mode support
    if (demoMode) {
      await executeDemoTrade(bestSignal);
      return;
    }
    const orderResult = await placeBinanceOrder(bestSignal, balance);
    if (orderResult) {
      bestSignal.orderId = orderResult.orderId;
      bestSignal.status = 'active';
      bestSignal.current_price = bestSignal.entry;
      bestSignal.roi = 0; bestSignal.tp1_hit = false; bestSignal.tp2_hit = false; bestSignal.tp3_hit = false;
      signals.push(bestSignal); positions.push(bestSignal);
      localStorage.setItem('sniper_signals', JSON.stringify(signals));
      localStorage.setItem('sniper_positions', JSON.stringify(positions));
      dailyCount.count++; lastTradeTime = Date.now();
      localStorage.setItem('sniper_daily', JSON.stringify(dailyCount));
      renderSignals(); updateStats(); renderPositions();
      showToast(`🎯 SNIPER SIGNAL: ${bestSignal.symbol} ${bestSignal.direction} @ $${bestSignal.entry.toFixed(2)}`);
      if (settings.telegramToken && settings.chatId) sendTelegramSignal(bestSignal);
    }
  }
  updateMoneyManagement(balance);
}

async function updatePositionPrices() {
  if (!currentUser) return;
  const activePositions = positions.filter(p => p.status === 'active' || p.status === 'tp1_hit' || p.status === 'tp2_hit');
  if (activePositions.length === 0) return;
  for (const pos of activePositions) {
    try {
      const price = await getSymbolPrice(pos.symbol);
      if (price) {
        pos.current_price = price;
        pos.roi = pos.direction === 'LONG' ? ((price - pos.entry) / pos.entry) * 100 : ((pos.entry - price) / pos.entry) * 100;
        if (!pos.tp1_hit && pos.direction === 'LONG' && price >= pos.tp1) {
          pos.tp1_hit = true; pos.status = 'tp1_hit';
          showToast(`🎯 TP1 HIT: ${pos.symbol} +${(pos.roi).toFixed(2)}%`);
          if (settings.telegramToken) sendTelegramTP(pos, 1);
        }
        if (!pos.tp2_hit && pos.direction === 'LONG' && price >= pos.tp2) {
          pos.tp2_hit = true; pos.status = 'tp2_hit';
          showToast(`🎯 TP2 HIT: ${pos.symbol} +${(pos.roi).toFixed(2)}%`);
          if (settings.telegramToken) sendTelegramTP(pos, 2);
        }
        if (pos.direction === 'LONG' && price >= pos.tp3) { closePosition(pos, 'tp3'); continue; }
        if (!pos.tp1_hit && pos.direction === 'SHORT' && price <= pos.tp1) {
          pos.tp1_hit = true; pos.status = 'tp1_hit';
          showToast(`🎯 TP1 HIT: ${pos.symbol} +${(pos.roi).toFixed(2)}%`);
          if (settings.telegramToken) sendTelegramTP(pos, 1);
        }
        if (!pos.tp2_hit && pos.direction === 'SHORT' && price <= pos.tp2) {
          pos.tp2_hit = true; pos.status = 'tp2_hit';
          showToast(`🎯 TP2 HIT: ${pos.symbol} +${(pos.roi).toFixed(2)}%`);
          if (settings.telegramToken) sendTelegramTP(pos, 2);
        }
        if (pos.direction === 'SHORT' && price <= pos.tp3) { closePosition(pos, 'tp3'); continue; }
        if (pos.direction === 'LONG' && price <= pos.sl) { closePosition(pos, 'sl'); continue; }
        if (pos.direction === 'SHORT' && price >= pos.sl) { closePosition(pos, 'sl'); continue; }
        const liqDistance = Math.abs(price - pos.liqPrice) / pos.entry;
        if (liqDistance < 0.005) {
          closePosition(pos, 'liq');
          showToast(`🚨 LIQUIDATION WARNING CLOSED: ${pos.symbol}`, 'error'); continue;
        }
      }
    } catch (e) { console.error('Price update error:', e); }
  }
  localStorage.setItem('sniper_positions', JSON.stringify(positions));
  renderPositions(); updateStats();
}

function closePosition(pos, reason) {
  pos.status = 'completed';
  pos.exit_price = pos.current_price;
  pos.exit_reason = reason;
  pos.exit_time = new Date().toISOString();
  const pnl = pos.direction === 'LONG' ? ((pos.exit_price - pos.entry) / pos.entry) * 100 : ((pos.entry - pos.exit_price) / pos.entry) * 100;
  pos.roi = pnl;
  const isWin = pnl > 0;
  stats.total++;
  if (isWin) { stats.wins++; consecutiveLosses = 0; }
  else { stats.losses++; consecutiveLosses++; dailyLoss += Math.abs(pnl); }
  stats.winRate = stats.total > 0 ? (stats.wins / stats.total) * 100 : 0;
  stats.totalPnL += pnl;
  trades.push({ symbol: pos.symbol, direction: pos.direction, entry: pos.entry, exit_price: pos.exit_price, status: isWin ? 'completed' : 'stopped', roi: pnl, confidence: pos.confidence, timestamp: pos.exit_time, reason });
  localStorage.setItem('sniper_stats', JSON.stringify(stats));
  localStorage.setItem('sniper_trades', JSON.stringify(trades));
  localStorage.setItem('sniper_positions', JSON.stringify(positions));
  renderPositions(); renderHistory(); updateStats();
  const emoji = reason === 'tp3' ? '🚀' : reason === 'sl' ? '⛔' : '🚨';
  showToast(`${emoji} Position closed: ${pos.symbol} ${pnl >= 0 ? '+' : ''}${pnl.toFixed(2)}%`);
  if (settings.telegramToken) sendTelegramClose(pos, reason, pnl);
  if (dailyLoss >= settings.maxDailyLoss) {
    showToast(`🛑 Max daily loss (${settings.maxDailyLoss}%) reached!`, 'error');
    stopTradingBot();
  }
}

function toggleCoin(btn) {
  const symbol = btn.dataset.symbol;
  const isSelected = btn.classList.contains('selected');
  const selectedCount = document.querySelectorAll('.coin-btn.selected').length;
  if (isSelected) {
    if (selectedCount <= 1) { showToast('❌ At least 1 coin required!', 'error'); return; }
    btn.classList.remove('selected');
    settings.selectedCoins = settings.selectedCoins.filter(c => c !== symbol);
  } else {
    if (selectedCount >= 6) { showToast('❌ Maximum 6 coins allowed!', 'error'); return; }
    btn.classList.add('selected');
    settings.selectedCoins.push(symbol);
  }
  localStorage.setItem('sniper_settings', JSON.stringify(settings));
}

  });
}

async function executeDemoTrade(signal) {
  if (!demoMode) return;

  const positionSize = Math.min(signal.positionSize, demoBalance * 0.2);
  const margin = positionSize / settings.leverage;

  if (margin > demoBalance * 0.95) {
    showToast('❌ ඩෙමෝ බැලන්ස් අප්‍රමාණවත්!', 'error');
    return;
  }

  const position = {
    ...signal,
    positionSize,
    margin,
    status: 'active',
    current_price: signal.entry,
    roi: 0,
    tp1_hit: false,
    tp2_hit: false,
    tp3_hit: false,
    demo: true
  };

  demoPositions.push(position);
  demoBalance -= margin;
  saveDemoData();
  updateDemoDisplay();

  showToast(`🎯 ඩෙමෝ ට්‍රේඩ්: ${signal.symbol} ${signal.direction} @ $${signal.entry.toFixed(2)}`);

  // Simulate price movement
  simulateDemoPriceMovement(position);
}

async function simulateDemoPriceMovement(position) {
  const interval = setInterval(async () => {
    if (!demoMode || position.status !== 'active') {
      clearInterval(interval);
      return;
    }

    const price = await getSymbolPrice(position.symbol);
    if (!price) return;

    position.current_price = price;
    position.roi = position.direction === 'LONG' 
      ? ((price - position.entry) / position.entry) * 100 
      : ((position.entry - price) / position.entry) * 100;

    // Check TP/SL
    if (position.direction === 'LONG') {
      if (price >= position.tp3) { closeDemoPosition(position, 'tp3'); clearInterval(interval); }
      else if (price <= position.sl) { closeDemoPosition(position, 'sl'); clearInterval(interval); }
    } else {
      if (price <= position.tp3) { closeDemoPosition(position, 'tp3'); clearInterval(interval); }
      else if (price >= position.sl) { closeDemoPosition(position, 'sl'); clearInterval(interval); }
    }

    saveDemoData();
    updateDemoDisplay();
  }, 5000);
}

function closeDemoPosition(position, reason) {
  position.status = 'completed';
  const pnl = position.roi;
  const profit = position.margin * (pnl / 100) * settings.leverage;
  demoBalance += position.margin + profit;

  demoStats.total++;
  if (pnl > 0) demoStats.wins++; else demoStats.losses++;
  demoStats.winRate = demoStats.total > 0 ? (demoStats.wins / demoStats.total) * 100 : 0;
  demoStats.totalPnL += profit;

  demoTrades.push({
    symbol: position.symbol,
    direction: position.direction,
    entry: position.entry,
    exit_price: position.current_price,
    status: pnl > 0 ? 'completed' : 'stopped',
    roi: pnl,
    confidence: position.confidence,
    timestamp: new Date().toISOString(),
    reason
  });

  demoPositions = demoPositions.filter(p => p !== position);
  saveDemoData();
  updateDemoDisplay();

  const emoji = reason === 'tp3' ? '🚀' : '⛔';
  showToast(`${emoji} ඩෙමෝ ට්‍රේඩ් අවසන්: ${position.symbol} ${pnl >= 0 ? '+' : ''}${pnl.toFixed(2)}%`);
}

function saveBotState() {
  localStorage.setItem('sniper_bot_state', JSON.stringify({
    running: botRunning,
    lastTradeTime: lastTradeTime,
    dailyLoss: dailyLoss,
    consecutiveLosses: consecutiveLosses,
    dailyCount: dailyCount,
    timestamp: new Date().toISOString()
  }));
}

function loadBotState() {
  const saved = localStorage.getItem('sniper_bot_state');
  if (saved) {
    const state = JSON.parse(saved);
    // Only restore if within same day
    const savedDate = new Date(state.timestamp).toISOString().split('T')[0];
    const today = new Date().toISOString().split('T')[0];

    if (savedDate === today) {
      lastTradeTime = state.lastTradeTime || 0;
      dailyLoss = state.dailyLoss || 0;
      consecutiveLosses = state.consecutiveLosses || 0;
      dailyCount = state.dailyCount || { count: 0, date: today };

      // If bot was running, restart it
      if (state.running && settings.apiKey && settings.secretKey) {
        setTimeout(() => startTradingBot(), 2000);
      }
    }
  }
}

async function demoScanAndTrade() {
  if (!demoBotRunning) return;

  const today = new Date().toISOString().split('T')[0];
  if (demoDailyCount.date !== today) {
    demoDailyCount = { count: 0, date: today };
    demoDailyLoss = 0;
    localStorage.setItem('sniper_demo_daily', JSON.stringify(demoDailyCount));
  }

  if (demoDailyCount.count >= settings.maxDailySignals) {
    showToast('📅 Demo daily trade limit reached (' + settings.maxDailySignals + ')');
    stopDemoTradingBot();
    return;
  }

  const now = Date.now();
  const cooldownMs = settings.cooldown * 60 * 1000;
  if (now - demoLastTradeTime < cooldownMs) return;

  const activePos = demoPositions.filter(p => p.status === 'active' || p.status === 'tp1_hit' || p.status === 'tp2_hit');
  if (activePos.length >= settings.maxPositions) return;

  if (demoBalance < settings.minBalance) {
    showToast('❌ Demo balance insufficient!', 'error');
    stopDemoTradingBot();
    return;
  }

  // Scan selected coins
  const coins = settings.selectedCoins || ['BTCUSDT', 'ETHUSDT'];
  let bestSignal = null;
  let bestConfluence = 0;

  for (const symbol of coins) {
    try {
      const signal = await analyzeSymbol(symbol);
      if (signal && signal.confluence > bestConfluence) {
        bestSignal = signal;
        bestConfluence = signal.confluence;
      }
    } catch (e) {
      console.error('Demo scan error for', symbol, e);
    }
  }

  if (bestSignal) {
    // Calculate demo position size
    const positionSize = Math.min(bestSignal.positionSize, demoBalance * 0.2);
    const margin = positionSize / settings.leverage;

    if (margin > demoBalance * 0.95) {
      showToast('❌ Demo balance අප්‍රමාණවත්!', 'error');
      return;
    }

    const position = {
      ...bestSignal,
      positionSize,
      margin,
      status: 'active',
      current_price: bestSignal.entry,
      roi: 0,
      tp1_hit: false,
      tp2_hit: false,
      tp3_hit: false,
      demo: true
    };

    demoPositions.push(position);
    demoBalance -= margin;

    demoDailyCount.count++;
    demoLastTradeTime = Date.now();
    localStorage.setItem('sniper_demo_daily', JSON.stringify(demoDailyCount));

    saveDemoData();
    updateDemoDisplay();
    renderSignals();
    updateStats();

    showToast(`🎯 DEMO SNIPER: ${bestSignal.symbol} ${bestSignal.direction} @ $${bestSignal.entry.toFixed(2)}`);

    // Start price simulation for this position
    demoSimulatePriceMovement(position);
  }
}

async function demoSimulatePriceMovement(position) {
  const interval = setInterval(async () => {
    if (!demoBotRunning || position.status !== 'active') {
      clearInterval(interval);
      return;
    }

    try {
      const price = await getSymbolPrice(position.symbol);
      if (!price) return;

      position.current_price = price;
      position.roi = position.direction === 'LONG' 
        ? ((price - position.entry) / position.entry) * 100 
        : ((position.entry - price) / position.entry) * 100;

      // Check TP1
      if (!position.tp1_hit && position.direction === 'LONG' && price >= position.tp1) {
        position.tp1_hit = true; position.status = 'tp1_hit';
        showToast(`🎯 Demo TP1 HIT: ${position.symbol} +${position.roi.toFixed(2)}%`);
      }
      if (!position.tp1_hit && position.direction === 'SHORT' && price <= position.tp1) {
        position.tp1_hit = true; position.status = 'tp1_hit';
        showToast(`🎯 Demo TP1 HIT: ${position.symbol} +${position.roi.toFixed(2)}%`);
      }

      // Check TP2
      if (!position.tp2_hit && position.direction === 'LONG' && price >= position.tp2) {
        position.tp2_hit = true; position.status = 'tp2_hit';
        showToast(`🎯 Demo TP2 HIT: ${position.symbol} +${position.roi.toFixed(2)}%`);
      }
      if (!position.tp2_hit && position.direction === 'SHORT' && price <= position.tp2) {
        position.tp2_hit = true; position.status = 'tp2_hit';
        showToast(`🎯 Demo TP2 HIT: ${position.symbol} +${position.roi.toFixed(2)}%`);
      }

      // Check TP3 - close position
      if (position.direction === 'LONG' && price >= position.tp3) {
        demoClosePosition(position, 'tp3');
        clearInterval(interval);
        return;
      }
      if (position.direction === 'SHORT' && price <= position.tp3) {
        demoClosePosition(position, 'tp3');
        clearInterval(interval);
        return;
      }

      // Check SL
      if (position.direction === 'LONG' && price <= position.sl) {
        demoClosePosition(position, 'sl');
        clearInterval(interval);
        return;
      }
      if (position.direction === 'SHORT' && price >= position.sl) {
        demoClosePosition(position, 'sl');
        clearInterval(interval);
        return;
      }

      saveDemoData();
      updateDemoDisplay();
      renderPositions();
    } catch (e) {
      console.error('Demo price update error:', e);
    }
  }, 5000);
}

function demoClosePosition(position, reason) {
  position.status = 'completed';
  const pnl = position.roi;
  const profit = position.margin * (pnl / 100) * settings.leverage;
  demoBalance += position.margin + profit;

  demoStats.total++;
  if (pnl > 0) {
    demoStats.wins++;
    demoConsecutiveLosses = 0;
  } else {
    demoStats.losses++;
    demoConsecutiveLosses++;
    demoDailyLoss += Math.abs(pnl);
  }
  demoStats.winRate = demoStats.total > 0 ? (demoStats.wins / demoStats.total) * 100 : 0;
  demoStats.totalPnL += profit;

  demoTrades.push({
    symbol: position.symbol,
    direction: position.direction,
    entry: position.entry,
    exit_price: position.current_price,
    status: pnl > 0 ? 'completed' : 'stopped',
    roi: pnl,
    confidence: position.confidence,
    timestamp: new Date().toISOString(),
    reason: reason,
    demo: true
  });

  demoPositions = demoPositions.filter(p => p !== position);
  saveDemoData();
  updateDemoDisplay();
  renderPositions();
  renderHistory();
  updateStats();

  const emoji = reason === 'tp3' ? '🚀' : '⛔';
  showToast(`${emoji} Demo Position Closed: ${position.symbol} ${pnl >= 0 ? '+' : ''}${pnl.toFixed(2)}%`);

  if (demoDailyLoss >= settings.maxDailyLoss) {
    showToast(`🛑 Demo max daily loss (${settings.maxDailyLoss}%) reached!`, 'error');
    stopDemoTradingBot();
  }
}

async function demoUpdatePositionPrices() {
  if (!demoBotRunning) return;

  const activePositions = demoPositions.filter(p => p.status === 'active' || p.status === 'tp1_hit' || p.status === 'tp2_hit');
  if (activePositions.length === 0) return;

  for (const pos of activePositions) {
    try {
      const price = await getSymbolPrice(pos.symbol);
      if (price) {
        pos.current_price = price;
        pos.roi = pos.direction === 'LONG' 
          ? ((price - pos.entry) / pos.entry) * 100 
          : ((pos.entry - price) / pos.entry) * 100;
      }
    } catch (e) {
      console.error('Demo price update error:', e);
    }
  }

  saveDemoData();
  updateDemoDisplay();
  renderPositions();
}

    balance: demoBalance,
    positions: demoPositions,
    trades: demoTrades,
    stats: demoStats,
    lastUpdate: new Date().toISOString()

function loadDemoData() {
  const saved = localStorage.getItem('sniper_demo_data');
  if (saved) {
    const data = JSON.parse(saved);
    demoBalance = data.balance || 1000;
    demoPositions = data.positions || [];
    demoTrades = data.trades || [];
    demoStats = data.stats || { total: 0, wins: 0, losses: 0, winRate: 0, totalPnL: 0 };
  } else {
    // Initialize with default values
    demoBalance = 1000;
    demoPositions = [];
    demoTrades = [];
    demoStats = { total: 0, wins: 0, losses: 0, winRate: 0, totalPnL: 0 };
  }
}

function updateDemoDisplay() {
  if (currentTradingMode !== 'demo') return;

  document.getElementById('walletBalance').textContent = '$' + demoBalance.toFixed(2);
  document.getElementById('walletAvailable').textContent = '$' + (demoBalance * 0.95).toFixed(2);
  document.getElementById('walletMargin').textContent = '$' + (demoPositions.reduce((sum, p) => sum + (p.margin || 0), 0)).toFixed(2);

  const unrealized = demoPositions.reduce((sum, p) => {
    if (p.status === 'active' || p.status === 'tp1_hit' || p.status === 'tp2_hit') {
      return sum + (p.margin * (p.roi || 0) / 100 * settings.leverage);
    }
    return sum;
  }, 0);

  document.getElementById('walletUnrealized').textContent = (unrealized >= 0 ? '+' : '') + '$' + unrealized.toFixed(2);
  document.getElementById('walletUnrealized').style.color = unrealized >= 0 ? 'var(--accent)' : 'var(--accent-red)';
  document.getElementById('walletMaintMargin').textContent = '$' + (demoPositions.reduce((sum, p) => sum + (p.margin || 0) * 0.1, 0)).toFixed(2);

  const riskRatio = demoBalance > 0 ? (demoPositions.reduce((sum, p) => sum + (p.margin || 0), 0) / demoBalance) * 100 : 0;
  document.getElementById('walletRiskRatio').textContent = riskRatio.toFixed(1) + '%';
  document.getElementById('walletRiskRatio').style.color = riskRatio > 80 ? 'var(--accent-red)' : riskRatio > 50 ? 'var(--accent-gold)' : 'var(--accent)';

  const riskFill = document.getElementById('riskMeterFill');
  riskFill.style.width = Math.min(riskRatio, 100) + '%';
  riskFill.className = 'risk-fill ' + (riskRatio > 80 ? 'risk-high' : riskRatio > 50 ? 'risk-medium' : 'risk-low');

  // Update stats
  document.getElementById('statPnL').textContent = '$' + demoStats.totalPnL.toFixed(2);
  document.getElementById('statWinRate').textContent = demoStats.winRate.toFixed(1) + '%';

  const activePos = demoPositions.filter(p => p.status === 'active' || p.status === 'tp1_hit' || p.status === 'tp2_hit');
  document.getElementById('statPositions').textContent = `${activePos.length}/${settings.maxPositions || 2}`;

  document.getElementById('statSignals').textContent = `${demoDailyCount.count}/${settings.maxDailySignals || 6}`;
  document.getElementById('signalProgress').style.width = `${(demoDailyCount.count / (settings.maxDailySignals || 6)) * 100}%`;

  // Update money management
  document.getElementById('mmBalance').textContent = '$' + demoBalance.toFixed(2);
  document.getElementById('mmMaxRisk').textContent = '$' + (demoBalance * (settings.riskPerTrade / 100)).toFixed(2);
  document.getElementById('mmMaxDailyLoss').textContent = '$' + (demoBalance * (settings.maxDailyLoss / 100)).toFixed(2);
  document.getElementById('mmCurrentLoss').textContent = '$' + (demoDailyLoss * (demoBalance / 100)).toFixed(2);
  document.getElementById('mmRemainingBudget').textContent = '$' + ((demoBalance * (settings.maxDailyLoss / 100)) - (demoDailyLoss * (demoBalance / 100))).toFixed(2);
}
